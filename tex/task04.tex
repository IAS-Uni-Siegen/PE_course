\task{Transistor-based AC/DC converter (single-phase AFE)}

\taskGerman{Transistorbasierter AC/DC-Umrichter}

An electric-drive test bench is connected to a single-phase grid via a full-bridge transistor AFE rectifier.
The converter should operate with approximately unity power factor and a stable DC-link voltage.
During testing, increased DC-link ripple is observed and the PWM strategy is questioned.

\begin{germanblock}
    Ein elektrischer Antriebsprüfstand ist über einen einphasigen Vollbrücken-Transistor-Gleichrichter
    an das Netz angeschlossen. Der Umrichter soll mit näherungsweise Leistungsfaktor Eins und stabiler
    DC-Zwischenkreisspannung betrieben werden. Während der Tests wird eine erhöhte Zwischenkreiswelligkeit
    beobachtet und die PWM-Strategie wird hinterfragt.
\end{germanblock}

\subtask{For approximately unity power factor operation, determine the required grid current RMS value and
    the corresponding grid current peak value.}{0}
\subtaskGerman{Bestimmen Sie für den Betrieb mit näherungsweise Leistungsfaktor Eins den erforderlichen
    Effektivwert des Netzstroms sowie den zugehörigen Spitzenwert des Netzstroms.}
\begin{solutionblock}
    \textbf{Auxiliary quantities}
    $$\hat{u}_g=\sqrt{2}U_g=\SI{325.27}{\volt},\qquad
        \omega=2\pi f=\SI{314.16}{\per\second}. $$

    \textbf{Grid current at unity power factor}
    $$P=U_g I_g \Rightarrow I_g=\frac{\SI{3000}{\watt}}{\SI{230}{\volt}}=\SI{13.04}{\ampere}.$$
    $$\hat{i}_g=\sqrt{2}I_g=\SI{18.44}{\ampere}. $$
\end{solutionblock}
\subtask{Calculate the required converter input-voltage amplitude $\hat{u}_2$ from the phasor relation and verify
    whether the nominal DC-link voltage is feasible.}{0}
\subtaskGerman{Berechnen Sie aus der komplexen Zeigerbeziehung die erforderliche Amplitude der Umrichtereingangsspannung $\hat{u}_2$
    und prüfen Sie, ob die nominelle DC-Zwischenkreisspannung ausreicht.}
\begin{solutionblock}
    \textbf{Required converter input-voltage amplitude}
    $$\hat{u}_2=\sqrt{\hat{u}_g^2+(\omega L\hat{i}_g)^2}
        =\sqrt{(325.27)^2+(314.16\cdot2.5\cdot10^{-3}\cdot18.44)^2}
        =\SI{325.59}{\volt}. $$
    With $U_{dc}=\SI{450}{\volt}>\hat{u}_2$, the boost rectifier condition is fulfilled.
\end{solutionblock}
\subtask{Classify the operating mode for each case below and justify using power-flow direction:}{0}
\begin{itemize}
    \item Case A: $P=+\SI{8}{\kilo\watt}$,
    \item Case B: $P=-\SI{5}{\kilo\watt}$.
\end{itemize}
\subtaskGerman{Ordnen Sie für die folgenden Fälle die Betriebsart zu und begründen Sie mit der Leistungsflussrichtung:
    \begin{itemize}
        \item Fall A: $P=+\SI{8}{\kilo\watt}$,
        \item Fall B: $P=-\SI{5}{\kilo\watt}$.
    \end{itemize}}
\begin{solutionblock}
    \textbf{Operating mode classification}
    \begin{itemize}
        \item Case A ($P=+\SI{8}{\kilo\watt}$): \textbf{Rectifier mode}, power flows from AC grid to DC link/load.
        \item Case B ($P=-\SI{5}{\kilo\watt}$): \textbf{Inverter mode}, power flows from DC side back to AC grid.
    \end{itemize}
\end{solutionblock}
\subtask{Determine the DC-link voltage oscillation amplitude
    $\hat{u}_C$ and the peak-to-peak ripple $\Delta U_{dc,pp}\approx2\hat{u}_C$.}{0}
\subtaskGerman{Bestimmen Sie die Amplitude der DC-Zwischenkreis-Spannungsoszillation $\hat{u}_C$ und den Peak-to-Peak-Wert $\Delta U_{dc,pp}\approx2\hat{u}_C$.}
\begin{solutionblock}
    \textbf{DC-link ripple estimate}
    $$Q_2=\omega L I_g^2=314.16\cdot2.5\cdot10^{-3}\cdot(13.04)^2=\SI{133.7}{\volt\ampere}.$$
    $$S_2=\sqrt{P^2+Q_2^2}=\sqrt{(3000)^2+(133.7)^2}=\SI{3003}{\volt\ampere}. $$
    From lecture Eq. (6.23), the DC-link current ripple amplitude is approximately
    $$\hat{i}_C\approx\frac{S_2}{U_{dc}}=\frac{3003}{450}=\SI{6.67}{\ampere}. $$
    Hence (Eq. (6.26))
    $$\hat{u}_C\approx\frac{\hat{i}_C}{2\omega C_{dc}}=\frac{6.67}{2\cdot314.16\cdot2.2\cdot10^{-3}}=\SI{4.82}{\volt},$$
    $$\Delta U_{dc,pp}\approx2\hat{u}_C=\SI{9.64}{\volt}. $$
\end{solutionblock}
\subtask{The engineering team suspects that the choice of PWM strategy contributes to the observed DC-link ripple.
    Using a sinusoidal reference $s^*(t)=\hat{s}\sin(\omega_1 t)$ and a triangular carrier $c(t)\in[-1,1]$, sketch the switching states of both complementary PWM and interleaved PWM over one fundamental period in the provided templates.
    For the shown switching interval, denote the local modulation value by $m=s^*(t_0)$ and identify the sequence of converter output states within $T_s$.
    Finally, explain which strategy produces a lower AC-side current ripple in $L$ and why this reduces stress on the DC-link capacitor $C_{\mathrm{dc}}$.}{0}
\subtaskGerman{Das Engineering-Team vermutet, dass die Wahl der PWM-Strategie zur beobachteten DC-Zwischenkreiswelligkeit beiträgt.
    Verwenden Sie ein sinusförmiges Referenzsignal $s^*(t)=\hat{s}\sin(\omega_1 t)$ und einen Dreiecksträger $c(t)\in[-1,1]$, und skizzieren Sie die Schaltzustände beider Strategien (Komplementär-PWM und Überlagerungs-PWM) über eine Schaltperiode $T_s$ in den bereitgestellten Vorlagen.
    Bezeichnen Sie für das gezeigte Schaltintervall den lokalen Modulationswert mit $m=s^*(t_0)$ und geben Sie die Folge der Umrichterausgangszustände innerhalb von $T_s$ an.
    Erläutern Sie abschließend, welche Strategie eine geringere netzstromseitige Stromwelligkeit in $L$ erzeugt und warum dies die Belastung des DC-Zwischenkreiskondensators $C_{\mathrm{dc}}$ verringert.}

\input{fig/Task04/Fig_subtask4.5_pwm_template_interleaved.tex}
\begin{solutionblock}
    \textbf{Given}
    The modulation reference is sinusoidal, $s^*(t)=\hat{s}\sin(\omega_1 t)$. For the plotted switching interval, denote the local value as $m=s^*(t_0)$.

    \textbf{1) Complementary PWM}
    $$s_1(t)=\begin{cases}
            +1, & m>c(t)    \\
            -1, & m\le c(t)
        \end{cases},\qquad s_2(t)=-s_1(t).$$
    The equivalent converter state is
    $$s_{\mathrm{cs}}(t)=\frac{s_1(t)-s_2(t)}{2}=s_1(t)\in\{+1,-1\}. $$

    \textbf{2) Interleaved PWM}
    $$s_1(t)=\begin{cases}
            +1, & m>c(t)    \\
            -1, & m\le c(t)
        \end{cases},\qquad
        s_2(t)=\begin{cases}
            +1, & -m>c(t)    \\
            -1, & -m\le c(t)
        \end{cases}. $$
    The equivalent three-level state is
    $$s_{\mathrm{is}}(t)=\frac{s_1(t)-s_2(t)}{2}\in\{+1,0,-1\}. $$
    A correct sketch over one $T_s$ shows transition instants at the intersections of $c(t)$ with $s^*(t)$ and $-s^*(t)$.

    \textbf{What must be visible in the plots}
    \begin{itemize}
        \item Complementary PWM: one carrier $c(t)$, one reference $s^*(t)$, states $s_1,s_2$, and resulting $s\in\{+1,-1\}$.
        \item Interleaved PWM: one carrier $c(t)$, both references $s^*(t)$ and $-s^*(t)$, states $s_1,s_2$, and resulting $s\in\{+1,0,-1\}$.
    \end{itemize}

    \input{fig/Task04/sFig_subtask4.5_pwm_solution_complementary.tex}
    \input{fig/Task04/sFig_subtask4.5_pwm_solution_interleaved.tex}

    \textbf{Comparison and impact on DC-link ripple}

    The worst cases are
    $$\max_m \Delta i_{L,\mathrm{cs},pp}=\frac{U_{dc}T_s}{2L} \quad \text{at } m=0,$$
    $$\max_m \Delta i_{L,\mathrm{is},pp}=\frac{U_{dc}T_s}{8L} \quad \text{at } |m|=\tfrac{1}{2},$$
    so the worst-case interleaved ripple is only one quarter of the worst-case complementary ripple.Since the high-frequency capacitor current is caused by the switching ripple power of the converter, the smaller inductor current ripple of interleaved PWM reduces the current stress of $C_{\mathrm{dc}}$ and therefore lowers the DC-link voltage ripple $\Delta U_{dc,pp}$ observed in the previous subtask.  Interleaved PWM is therefore the preferable strategy.
    \vfill
\end{solutionblock}
